    function verificaRisposte() {

      const soluzioni = ["P", "E", "U", "G", "E", "O", "T", "A", "C", "Q", "U", "A", "C", "A", "P", "U", "T", "O", "P", "A","L", "E", "S", "T", "R", "A", "O", "R", "Z", "A", "T", "A", "G", "E", "O", "M", "E", "T", "R", "I", "A", "A", "B", "I", "T", "O", "L", "U", "N", "A", "N", "I", "C", "H", "O", "L","A", "S", "N", "O", "V", "A", "N", "T", "A", "D", "U", "E"];


      const inputElements = document.querySelectorAll('input[type="text"]');

      let tutteCorrette = true;


      inputElements.forEach((inputElement, index) => {

        const soluzioneCorretta = soluzioni[index];

        const letteraInserita = inputElement.value.toUpperCase();


        if (letteraInserita === soluzioneCorretta) {

          inputElement.style.backgroundColor = 'green';

        } else {

          inputElement.style.backgroundColor = 'red';

          tutteCorrette = false;

        }

      });


    if (tutteCorrette) {

      alert('Complimenti! Hai completato il cruciverba correttamente.');


      // Lista delle posizioni (riga, colonna) delle caselle da animare

      const caselleDaAnimare = [

        [1, 1],

        [2, 5],

        [3, 3],

        [4, 1],

        [5, 4],

        [6, 1],

        [7, 1],

        [8, 1],

        [9, 6],

        [10, 2]

      ];


      // Aggiungi classe .animate-cell alle caselle da animare

      caselleDaAnimare.forEach((pos) => {

        const [riga, colonna] = pos;

        const casellaDaAnimare = document.querySelector(`tr:nth-child(${riga}) td:nth-child(${colonna})`);

        casellaDaAnimare.classList.add('animate-cell');

      });

    } else {

      alert('Ci sono degli errori. Controlla le tue risposte e riprova.');

    }

  }

</script>
